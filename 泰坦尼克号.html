#四、选择最佳参数mtry (10折交叉验证)、并用验证集验证
# 1、增加验证集(从train1中划分)
set.seed(500)
train_indexes=createDataPartition(train1$Survived, times=1, p=0.8, list=FALSE)      #createDataPartition()用于等比随机抽样

tr_train = train1[train_indexes,]
tr_test = train1[-train_indexes,]



# 2、10折交叉验证准备工作
# (1)tr_train平均分10组
set.seed(2348)  
folds = createMultiFolds(tr_train$Survived,  k = 10, times = 10)    #确定分组：将tr_train均分成10组，每组又分10次

# (2)确定进行几次，几折交叉验证
my_control = trainControl(method = "repeatedcv", number = 10, repeats = 10,  index = folds) #确定对folds进行10次10折交叉验证



# 3、进行10交叉验证 (用随机森林的方法)
set.seed(1234)
RF3 = train(x = tr_train[,-1], y = tr_train[,1], method = "rf", trControl =my_control)     #进行随机森林方式的交叉验证
RF3

#选择最佳参数mtry=2    准确率 83.75%



# 4、用(验证集)tr_test验证(选定的最佳模型)
pred_RF3 = predict(RF3 , tr_test)
confusionMatrix(pred_RF3 , tr_test$Survived)    #准确率 85.31%

4.3 逻辑回归

#一、增加验证集 (从train中划分)
library(caret)
set.seed(500)
train_indexes=createDataPartition(train$Survived, times=1, p=0.8, list=FALSE)      #createDataPartition()用于随机抽样

tr_train = train[train_indexes,]
tr_test = train[-train_indexes,]





#二、建立逻辑回归模型 (用tr_train)
lr = glm(Survived ~ ., family = binomial(link=logit), data = tr_train)

summary(lr)
confint(lr)





#三、验证准确率

#1、预测tr_train
prediction_train_lr = predict(lr, data=tr_train, type ="response")

table(tr_train$Survived, prediction_train_lr>0.5)   #tr_train的混淆矩阵
(193+396)/(193+396+73+51)    #准确率 82.6%


#2、预测tr_test
prediction_test_lr = predict(lr, newdata=tr_test, type ="response")       #注意这里用newdata，不然data不变

table(tr_test$Survived, prediction_test_lr>0.5)     #tr_test的混淆矩阵
(59+91)/(59+91+17+11)        #准确率 84.27%
